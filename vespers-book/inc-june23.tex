\section{June 23: Dedication of Birmingham Oratory Church}
\subtitle{\nth{1} Class}
\medskip
\printnote{If today is Sunday, 23 June then \emph{Dedication Vespers} will not be sung but \emph{First Vespers of the Nativity of St John the Baptist}, which follows, take precedence with a commemoration of \emph{Second Vespers of the Dedication}.

If today is Sunday, 22 June then Vespers will be \emph{First Vespers of the Dedication}, on page \pageref{commondedicationofchurch}, without a commemoration of the Sunday.}
