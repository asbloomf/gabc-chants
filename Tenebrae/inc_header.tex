\newcommand{\mydropcap}[3][latin]{%
% 1 language
% 2 first word or most of text
% 3 rest of text
  \StrPosition{#2}{ }[\mypos]%
  \ifnum\mypos=2
  % find the next space then
    \StrPosition[2]{#2}{ }[\mypos]%
  \fi
  \ifnum\mypos=0
    \StrLen{#2}[\mypos]
    \setcounter{mydccounter}{\mypos}
  \else
    \setcounter{mydccounter}{\mypos - 1}
  \fi
  \StrLeft{#2}{1}[\asbdropcapletter]
  \StrMid{#2}{2}{\themydccounter}[\asbdropcapword]
  \StrGobbleLeft{#2}{\mypos}[\asbdropcappar]
  \lettrine{\asbdropcapletter}%
     {\asbdropcapword}%
  {%
    \selectlanguage{#1}
    \asbdropcappar #3
  }%
}
